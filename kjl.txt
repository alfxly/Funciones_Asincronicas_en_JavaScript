<script>
async function calcularPromedio() {
  const num1 = document.getElementById('numero1').value;
  const num2 = document.getElementById('numero2').value;
  const num3 = document.getElementById('numero3').value;
  const resultado = await obtenerPromedio(num1, num2, num3);
  document.getElementById('resultado').innerText = resultado;
}

function obtenerPromedio(num1, num2, num3) {
  return new Promise(resolve => {
    setTimeout(() => {
      num1 = Number(num1);
      num2 = Number(num2);
      num3 = Number(num3);
      if (!num1 || !num2 || !num3) {
        resolve('Por favor, ingresa tres números válidos.');
      } else {
        const promedio = (num1 + num2 + num3) / 3;
        resolve(`El promedio es: ${promedio.toFixed(2)}.`);
      }
    }, 1000); // Simula un cálculo asincrónico con un retraso
  });
}
</script>
